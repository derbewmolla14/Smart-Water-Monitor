### 2. በ Backend ላይ የዳታ ማውረጃ መንገድ (Route) መፍጠር
በ `server.js` ፋይልህ ውስጥ የሚከተለውን ኮድ ጨምር። ይህ ኮድ ከዳታቤዝ መረጃውን አውጥቶ ወደ Excel ይቀይረዋል።

```javascript
const ExcelJS = require('exceljs');

app.get('/download-excel', async (req, res) => {
    try {
        // 1. ሁሉንም የውሃ መጠን መረጃዎች ከዳታቤዝ አውጣ
        const data = await WaterLog.find().sort({ timestamp: -1 });

        // 2. አዲስ የ Excel ፋይል (Workbook) ፍጠር
        let workbook = new ExcelJS.Workbook();
        let worksheet = workbook.addWorksheet('Water Logs');

        // 3. የሰንጠረዡን አርዕስት (Headers) አዘጋጅ
        worksheet.columns = [
            { header: 'ቀን እና ሰዓት', key: 'timestamp', width: 25 },
            { header: 'የውሃ መጠን (%)', key: 'level', width: 15 },
            { header: 'ሁኔታ', key: 'status', width: 15 }
        ];

        // 4. መረጃዎቹን ወደ ሰንጠረዡ ጨምር
        data.forEach(log => {
            worksheet.addRow({
                timestamp: log.timestamp.toLocaleString('am-ET'),
                level: log.level,
                status: log.level < 20 ? 'ዝቅተኛ' : 'በቂ'
            });
        });

        // 5. ፋይሉን ለተጠቃሚው ላክ
        res.setHeader('Content-Type', 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet');
        res.setHeader('Content-Disposition', 'attachment; filename=' + 'WaterMonitor_Report.xlsx');

        await workbook.xlsx.write(res);
        res.end();

    } catch (error) {
        res.status(500).send("Excel ለማውረድ ሲሞከር ስህተት ተፈጥሯል");
    }
});
```

ESP32 ሴንሰሩን ተጠቅሞ የውሃውን ርቀት ይለካል፣ ከዚያም በ Wi-Fi አማካኝነት ወደ ሰርቨርህ ዳታውን ይልካል።

### 1. የሴንሰር ግንኙነት (Circuit Connection)
**Ultrasonic Sensor (HC-SR04)** አራት እግሮች አሉት። ከ ESP32 ጋር በሚከተለው መልኩ አገናኘው፦

* **VCC** -> ESP32 **5V** ወይም **Vin**
* **GND** -> ESP32 **GND**
* **Trig** -> ESP32 **Pin 5**
* **Echo** -> ESP32 **Pin 18**



### 2. የ ESP32 ኮድ (C++)
ይህ ኮድ በየ 5 ሰከንዱ የውሃውን ደረጃ ለክቶ ወደ ሰርቨርህ ይልካል። በ Arduino IDE ወይም VS Code (PlatformIO) መጠቀም ትችላለህ።

```cpp



const char* ssid = "የWi-Fi_ስም";
const char* password = "የWi-Fi_ፓስዎርድ";

// የኮምፒውተርህ IP አድራሻ (ለምሳሌ 192.168.1.5)
const char* serverName = "http://192.168.1.5:3000/update-level";

const int trigPin = 5;
const int echoPin = 18;

void setup() {
  Serial.begin(115200);
  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);

  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.println("ከWi-Fi ጋር እየተገናኘሁ ነው...");
  }
  Serial.println("ተገናኝቷል!");
}

void loop() {
  // 1. ርቀቱን መለካት
  long duration;
  float distance;
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);
  
  duration = pulseIn(echoPin, HIGH);
  distance = (duration * 0.034) / 2; // ወደ ሴንቲሜትር መቀየር

  // 2. ርቀቱን ወደ ፐርሰንት መቀየር (ለምሳሌ የታንከሩ ቁመት 100 ሳ.ሜ ቢሆን)
  int waterPercent = map(distance, 100, 0, 0, 100); 
  if(waterPercent > 100) waterPercent = 100;
  if(waterPercent < 0) waterPercent = 0;

  // 3. ወደ ሰርቨር መላክ
  if (WiFi.status() == WL_CONNECTED) {
    HTTPClient http;
    String url = String(serverName) + "?level=" + String(waterPercent);
    http.begin(url);
    int httpResponseCode = http.GET();
    
    if (httpResponseCode > 0) {
      Serial.print("ዳታ ተልኳል! Response: ");
      Serial.println(httpResponseCode);
    }
    http.end();
  }
  
  delay(5000); // በየ 5 ሰከንዱ ይለካል
}
```
